 7 import java.util.Scanner;
 8 import java.time.temporal.ValueRange;
 9 
10 @FunctionalInterface
11 interface SwitchGrading{
12     abstract String grade(int mark);
13 }
14 
15 public class SwitchGrader {
16     public static void main(String args[]) {
17         
18         SwitchGrading grader = (int mark) -> {
19             ValueRange gradeA = ValueRange.of(80, 100);
20             ValueRange gradeB = ValueRange.of(60, 79);
21             ValueRange gradeC = ValueRange.of(40, 59);
22             ValueRange gradeD = ValueRange.of(30, 39);
23             ValueRange fail = ValueRange.of(0, 29);
24             switch(
25                     (gradeA.isValidValue(mark)) ? "A" : 
26                     (gradeB.isValidValue(mark)) ? "B" : 
27                     (gradeC.isValidValue(mark)) ? "C" : 
28                     (gradeD.isValidValue(mark)) ? "D" : 
29                     (fail.isValidValue(mark)) ? "F" : "INVALID"
30                     ){
31                 case "A":
32                     return "A";
33                 case "B":
34                     return "B";
35                 case "C":
36                     return "C";
37                 case "D":
38                     return "D";
39                 case "F":
40                     return "FAIL";
41                 default:
42                     return "INVALID";
43             }            
44         };
45         
46         Scanner input = new Scanner(System.in);
47         System.out.print("\n --- System Online [Ready to Grade] --- ");
48         while(true){
49             try{
50                 System.out.print("\n\t Enter mark: ");
51                 int mark = input.nextInt();
52                 System.out.println("\t\t grade: " + grader.grade(mark));
53             }catch(Exception e){
54                 break;
55             }
56         }
57     }
58 }

Question:

	Write an application program in java to convert several temperatures in Fahrenheits to Celcius using the following conversion formula.

	C = (f - 32) / 1.8 and display the values in a tabular form.
 

 6 import java.util.ArrayList;
 7 
 8 public class QuestionFour {
 9     
10     public static ArrayList<Float> converter(Float... fahrenheits){
11         ArrayList<Float> celcius = new ArrayList<>();
12         for(Float f: fahrenheits){
13             Float result = (float) ((f - 32) / 1.8);
14             celcius.add(result);
15         }
16         printTable(fahrenheits, celcius);
17         return celcius;
18     }
19     
20     public static void printTable(Float[] fahrenheits, ArrayList celcius){
21         try{
22             assert(fahrenheits.length == celcius.size());
23             System.out.println("\n\n\t| *Fahrenheits \t | \t *Celcius \n" + 
24                     "\t-----------------------------------");
25             for(short i = 0; i < fahrenheits.length; i++){
26                 Float f = (float) fahrenheits[i];
27                 Float c = (float) celcius.get(i);
28                 System.out.println("\t| " + f + " \t\t | \t " + c);
29             }
30         }catch(Exception e){
31             System.out.println("--null output--");
32         }
33     }
34

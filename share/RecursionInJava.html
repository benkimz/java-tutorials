 5 public class RecursionInJava{
 6     // A SIMPLE RECURSIVE METHOD TO GET THE FACTORIAL OF A NUMBER
 7     public static int getTheSum(int n){
 8         int sum = 0;
 9         if(n > 0) sum += n + getTheSum(n - 1);
10         return sum;
11     }
12     // A SIMPLE RECURSIVE METHOD TO GET THE FACTORIAL OF A NUMBER
13     public static long getFactorialN(int n){
14         long factorial = 1;
15         if(n > 0) factorial *= n * getFactorialN(n - 1);
16         return factorial;
17     }
